    scripture_text += verse[0]
    if verse[1] != last_book:
        scripture_text += '$'
        last_book = verse[1]
    if verse[2] != last_chapter:
        scripture_text += '%'
        last_chapter = verse[2]
    if verse[3] != last_verse:
        scripture_text += '^'
        last_verse = verse[3]
